  <script>
    // Workout & Diet Plan
    function generatePlan() {
      let goal = document.getElementById("goal").value;
      let planText = "";

      if (goal === "lose") {
        planText = `
          <b>Workout:</b> 30 min Cardio + Light Strength<br>
          <b>Diet:</b> Eat more veggies, fruits, lean protein, less sugar.
        `;
      } else if (goal === "maintain") {
        planText = `
          <b>Workout:</b> 4 days/week mix of cardio & strength<br>
          <b>Diet:</b> Balanced meals (protein, carbs, fats) in moderation.
        `;
      } else if (goal === "gain") {
        planText = `
          <b>Workout:</b> 5 days Strength Training + Protein focus<br>
          <b>Diet:</b> High protein, whole grains, healthy fats, surplus calories.
        `;
      }

      document.getElementById("plan").innerHTML = planText;
    }

    // Order Items
    function placeOrder() {
      let checkboxes = document.querySelectorAll('.item-list input[type="checkbox"]');
      let selected = [];
      let prices = { "Protein Powder":1200, "Creatine":800, "Multivitamin":500, "Omega 3":600 };
      let total = 0;

      checkboxes.forEach(cb => {
        if(cb.checked){
          selected.push(cb.value);
          total += prices[cb.value];
        }
      });

      if(selected.length > 0){
        document.getElementById("cart").innerHTML = 
          "<b>You ordered:</b> " + selected.join(", ") + "<br><b>Total:</b> ₹" + total;
      } else {
        document.getElementById("cart").innerHTML = "No items selected.";
      }
    }
  </script>
